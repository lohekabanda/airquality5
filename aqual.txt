col_name = ['so2','no2','rspm','spm']
def remove_outliers(col):
    q1=col.quantile(0.25)
    q3=col.quantile(0.75)
    iqr=q3-q1
    threshold=1.5*iqr
    outiler_mask =(col<q1-threshold) | (col >q3+threshold)
    return col[~outiler_mask]

for col in col_name:
    df[col]=remove_outliers(df[col])
